CREATE VIEW v_auth AS (
 SELECT a.entity_id,
    a.password_hash,
    a.is_banned,
    a.role_id,
    r.name AS role_name,
    r.permissions AS role_permissions
   FROM (identity.auth a
     JOIN identity.role r ON ((r.id = a.role_id)))
)
